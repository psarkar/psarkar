\begin{align}
\theta(F)&=\int(1-F(x)-F(-x))^2dF(x)\\
&=\int((1-F(-x))^2dF(x)-2\int(1-F(-x))F(x)dF(x)+\int F(x)^2dF(x)
\end{align}
