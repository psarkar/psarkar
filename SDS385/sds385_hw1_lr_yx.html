def get_indicator(y, C):
    # get one-hot of categories
    I = np.zeros((len(y), C))
    for c in range(C):
        I[:,c] = (y == c)
    return I

def softmax(y_linear):
    exp = np.exp(y_linear)
    norms = np.sum(exp, axis=1).reshape((-1,1))
    return exp / norms

def cross_entropy(yhat, y):
    return - np.sum(y * np.log(yhat+1e-6))

def get_loss(X, I, beta):
    N = X.shape[0]
    y_linear = np.dot(X, beta)
    y_hat = softmax(y_linear)
    loss = 1.0/N * cross_entropy(y_hat, I)
    return loss

def gradient_descent(beta, pre_beta, X, I, eta=1, batch=1, theta=0):
    N = X.shape[0]
    grad = 1.0/N*np.dot(X.T, softmax(np.dot(X, beta))-I) + 2.0/N*beta
    beta = beta - eta*grad
    beta[:, -1] = 0
    return beta, None

def stochastic(beta, pre_beta, X, I, eta=1e-3, batch=1, theta=0):
    N = X.shape[0]
    idx = np.random.randint(N, size=1)
    grad = np.dot(X[idx,:].T, softmax(np.dot(X[idx,:], beta))-I[idx, :]) + 2.0/N*beta
    beta = beta - eta*grad
    beta[:, -1] = 0
    return beta, None

def mini_batch(beta, pre_beta, X, I, eta=1e-3, batch=16, theta=0):
    N = X.shape[0]
    idx = np.random.choice(N, batch, replace=False)
    grad = 1.0/batch*np.dot(X[idx,:].T, softmax(np.dot(X[idx,:], beta))-I[idx, :]) + 2.0/N*beta
    beta = beta - eta*grad
    beta[:, -1] = 0
    return beta, None

def momentum(beta, pre_beta, X, I, eta=1, batch=1, theta=0.1):
    N = X.shape[0]
    grad = 1.0/N*np.dot(X.T, softmax(np.dot(X, beta)) -I) + 2.0/N*beta
    beta_new = beta - eta*grad + theta*(beta - pre_beta)
    beta_new[:, -1] = 0
    return beta_new, beta

def newton(beta, pre_beta, X, I, eta=1, batch=1, theta=0):
    N = X.shape[0]
    n = beta.shape[0]
    C = beta.shape[1]
    grad = (1.0/N*np.dot(X.T, softmax(np.dot(X, beta))-I) + 2.0/N*beta)[:,:-1].reshape(-1,1,order='F')
    assert grad.shape[0] == n*(C-1)
    P = softmax(np.dot(X, beta))
    assert P.shape[0] == N
    hessian =  np.diag(2.0/N*np.ones(n*(C-1)))
    for i in range(N):
        xx = np.outer(X[i,:], X[i,:])
        p = P[i,:-1]
        pp = np.diag(p) - np.outer(p, p)
        hessian = hessian + 1.0/N*np.kron(pp, xx)
       
    beta_vec = beta[:,:-1].reshape(-1,1,order='F')
    assert beta_vec.shape[0] == n*(C-1)
    beta_vec = beta_vec - np.dot(inv(hessian), grad)
    beta[:,:-1] = beta_vec.reshape(n, C-1, order='F')
    beta[:, -1] = 0
    assert beta.shape[1] == C
    return beta, None

def descent(update, train, test, C, eta, batch=64, theta=0.8, gap=100, dec=0, T=int(1e3)):
    
    # split data from dataset
    X_train, y_train, X_test, y_test = data_label_split(train, test)
    
    # define parameters
    n = X_train.shape[1]
    
    # initialization
    np.random.seed(1023)
    # beta = np.random.rand(n, C)
    beta = np.zeros((n, C))
    ## let beta_C = 0
    beta[:, -1] = 0
    pre_beta = beta
    ## get one-hot category
    I_train = get_indicator(y_train, C)
    I_test = get_indicator(y_test, C)
    eta0 = eta
    
    train_err = []
    test_err = []
    
    for t in range(T):
        
        if (t % 1 == 0) or (t == T - 1):
            train_err.append(get_loss(X_train, I_train, beta))
            test_err.append(get_loss(X_test, I_test, beta))
 
            assert not np.isnan(train_err[-1])
            assert not np.isnan(test_err[-1])
            
        if (t % gap == 0) or (t == T - 1):
            print("iteration = {}: train loss = {}, test loss = {}".format(t, train_err[t], test_err[t]))
            
        if dec==1:
            eta = eta0 / (np.sqrt(t)+1.0)
        elif dec==2:
            eta = eta0 / (t+ 1.0)
            
        beta, pre_beta = update(beta, pre_beta, X_train, I_train, eta, batch, theta)
        assert beta[:, -1].all() == 0
            
    return beta, train_err, test_err

def get_plot(err, if_train, method, dataset):
    plt.clf()

    else:

    plt.xlabel('iteration')
    plt.ylabel('log error')

def newton_diag(beta, pre_beta, X, I, eta=1, batch=1, theta=0):
    N = X.shape[0]
    n = beta.shape[0]
    C = beta.shape[1]
    grad = 1.0/N*np.dot(X.T, softmax(np.dot(X, beta))-I) + 2.0/N*beta
    P = softmax(np.dot(X, beta))
    assert P.shape[0] == N
    assert P.shape[1] == C
    
    for i in range(C-1):
        hess = np.dot(X.T, np.dot(np.diag(P[:,i]-P[:,i]*P[:,i]), X)) + np.diag(2.0/N*np.ones(n))
        beta[:,i] = beta[:,i] - np.dot(inv(hess), grad[:,i])
    
    return beta, None
